## 2. CheatSheet模组建筑功能分析

### 2.1 模组概述

CheatSheet是泰拉瑞亚最受欢迎的辅助工具模组之一，提供了丰富的建筑生成功能。

**主要建筑功能：**
- **建筑模板系统**：预设多种建筑模板

### 2.3 模板放置核心代码

```csharp
// 模板放置方法（参考实现）
public void PlaceTemplate(int startX, int startY, BuildingTemplate template)
{
    // 1. 清理区域
    for (int x = startX; x < startX + template.Width; x++)
    {
        for (int y = startY; y < startY + template.Height; y++)
        {
            if (WorldGen.InWorld(x, y))
            {
                WorldGen.EmptyTile(x, y, true);
            }
        }
    }

    // 2. 放置Tile
    for (int x = 0; x < template.Width; x++)
    {
        for (int y = 0; y < template.Height; y++)
        {
            int worldX = startX + x;
            int worldY = startY + y;
            
            TemplateTile tile = template.Tiles[x][y];
            
            if (tile.TileType > 0 && WorldGen.InWorld(worldX, worldY))
            {
                WorldGen.PlaceTile(worldX, worldY, tile.TileType);
                
                // 设置油漆
                if (tile.TileColor > 0)
                {
                    Main.tile[worldX, worldY].TileColor = tile.TileColor;
                }
            }
        }
    }

    // 3. 放置Wall
    for (int x = 0; x < template.Width; x++)
    {
        for (int y = 0; y < template.Height; y++)
        {
            int worldX = startX + x;
            int worldY = startY + y;
            
            TemplateWall wall = template.Walls[x][y];
            
            if (wall.WallType > 0 && WorldGen.InWorld(worldX, worldY))
            {
                WorldGen.PlaceWall(worldX, worldY, wall.WallType);
                
                if (wall.WallColor > 0)
                {
                    Main.tile[worldX, worldY].WallColor = wall.WallColor;
                }
            }
        }
    }

    // 4. 放置家具
    foreach (var furniture in template.Furniture)
    {
        int worldX = startX + furniture.X;
        int worldY = startY + furniture.Y;
        
        if (WorldGen.InWorld(worldX, worldY))
        {
            WorldGen.PlaceObject(worldX, worldY, furniture.FurnitureType, false, furniture.Style);
        }
    }

    // 5. 同步到客户端（多人游戏）
    if (Main.netMode == NetmodeID.Server)
    {
        NetMessage.SendTileSquare(-1, startX, startY, template.Width, template.Height);
    }
}
```

```csharp
// 标准NPC房屋生成（推断实现）
public void GenerateNPCHouse(int x, int y, int style = 0)
{
    // 房屋尺寸配置
    int width = 10;
    int height = 8;

    // 根据样式选择材料
    ushort wallTile = GetWallTileByStyle(style);     // 外墙Tile
    ushort wallType = GetWallTypeByStyle(style);     // 背景墙
    ushort floorTile = GetFloorTileByStyle(style);   // 地板Tile

    // 1. 清理空间
    ClearArea(x, y, width, height);

    // 2. 建造框架
    // 地板
    for (int i = x; i < x + width; i++)
    {
        WorldGen.PlaceTile(i, y + height - 1, floorTile);
    }
    // 天花板
    for (int i = x; i < x + width; i++)
    {
        WorldGen.PlaceTile(i, y, wallTile);
    }
    // 左墙
    for (int j = y; j < y + height; j++)
    {
        WorldGen.PlaceTile(x, j, wallTile);
    }
    // 右墙
    for (int j = y; j < y + height; j++)
    {
        WorldGen.PlaceTile(x + width - 1, j, wallTile);
    }

    // 3. 放置背景墙
    for (int i = x + 1; i < x + width - 1; i++)
    {
        for (int j = y + 1; j < y + height - 1; j++)
        {
            WorldGen.PlaceWall(i, j, wallType);
        }
    }

    // 4. 放置门
    WorldGen.PlaceDoor(x + width / 2, y + height - 4, TileID.ClosedDoor);

    // 5. 放置必要家具
    // 工作台（平坦表面）
    WorldGen.PlaceObject(x + 2, y + height - 2, TileID.WorkBench);
    // 椅子（舒适物品）
    WorldGen.PlaceObject(x + 4, y + height - 2, TileID.Chairs);
    // 火把（光源）
    WorldGen.PlaceObject(x + 6, y + height - 3, TileID.Torches);

    // 6. 同步
    SyncArea(x, y, width, height);
}

// 样式对应的材料
private ushort GetWallTileByStyle(int style)
{
    switch (style)
    {
        case 0: return TileID.Wood;           // 木屋
        case 1: return TileID.StoneSlab;      // 石屋
        case 2: return TileID.Brick;          // 砖屋
        case 3: return TileID.GrayBrick;      // 灰砖屋
        case 4: return TileID.GoldBrick;      // 金砖屋
        default: return TileID.Wood;
    }
}
```

### 3.2 Structure Builder实现

HEROsMod的Structure Builder功能核心实现：

```csharp
// 结构生成器核心类
public class StructureBuilderTool
{
    // 当前选中的结构模板
    private StructureTemplate currentTemplate;
    
    // 放置模式
    public enum PlacementMode
    {
        Single,       // 单个放置
        Line,         // 线性放置
        Rectangle,    // 矩形填充
        Circle        // 圆形填充
    }

    // 线性放置方法
    public void PlaceLine(int startX, int startY, int endX, int endY, ushort tileType)
    {
        // 计算线性路径
        int dx = Math.Abs(endX - startX);
        int dy = Math.Abs(endY - startY);
        int sx = startX < endX ? 1 : -1;
        int sy = startY < endY ? 1 : -1;
        int err = dx - dy;

        int x = startX;
        int y = startY;

        while (true)
        {
            // 放置Tile
            if (WorldGen.InWorld(x, y))
            {
                WorldGen.PlaceTile(x, y, tileType);
            }

            if (x == endX && y == endY)
                break;

            int e2 = 2 * err;
            if (e2 > -dy)
            {
                err -= dy;
                x += sx;
            }
            if (e2 < dx)
            {
                err += dx;
                y += sy;
            }
        }

        // 同步
        SyncModifiedArea(startX, startY, endX, endY);
    }

    // 矩形填充方法
    public void PlaceRectangle(int startX, int startY, int width, int height, ushort tileType)
    {
        for (int x = startX; x < startX + width; x++)
        {
            for (int y = startY; y < startY + height; y++)
            {
                if (WorldGen.InWorld(x, y))
                {
                    WorldGen.PlaceTile(x, y, tileType);
                }
            }
        }

        SyncArea(startX, startY, width, height);
    }

    // 圆形填充方法
    public void PlaceCircle(int centerX, int centerY, int radius, ushort tileType, bool filled = true)
    {
        for (int x = centerX - radius; x <= centerX + radius; x++)
        {
            for (int y = centerY - radius; y <= centerY + radius; y++)
            {
                int dx = x - centerX;
                int dy = y - centerY;
                int distanceSq = dx * dx + dy * dy;

                if (filled)
                {
                    if (distanceSq <= radius * radius && WorldGen.InWorld(x, y))
                    {
                        WorldGen.PlaceTile(x, y, tileType);
                    }
                }
                else
                {
                    // 只画边缘
                    if (Math.Abs(distanceSq - radius * radius) < radius && WorldGen.InWorld(x, y))
                    {
                        WorldGen.PlaceTile(x, y, tileType);
                    }
                }
            }
        }

        SyncArea(centerX - radius, centerY - radius, radius * 2 + 1, radius * 2 + 1);
    }
}
```

### 3.3 Copy/Paste功能实现

```csharp
// 复制粘贴系统
public class StructureCopyPaste
{
    // 复制的数据结构
    private StructureData copiedData;

    // 复制结构
    public void CopyStructure(int startX, int startY, int width, int height)
    {
        copiedData = new StructureData();
        copiedData.Width = width;
        copiedData.Height = height;
        copiedData.Tiles = new TileData[width][];
        copiedData.Walls = new WallData[width][];

        for (int x = 0; x < width; x++)
        {
            copiedData.Tiles[x] = new TileData[height];
            copiedData.Walls[x] = new WallData[height];

            for (int y = 0; y < height; y++)
            {
                int worldX = startX + x;
                int worldY = startY + y;

                if (WorldGen.InWorld(worldX, worldY))
                {
                    Tile tile = Main.tile[worldX, worldY];

                    // 保存Tile数据
                    copiedData.Tiles[x][y] = new TileData
                    {
                        HasTile = tile.HasTile,
                        TileType = tile.TileType,
                        FrameX = tile.TileFrameX,
                        FrameY = tile.TileFrameY,
                        TileColor = tile.TileColor,
                        IsHalfBlock = tile.IsHalfBlock,
                        Slope = tile.Slope
                    };

                    // 保存Wall数据
                    copiedData.Walls[x][y] = new WallData
                    {
                        WallType = tile.WallType,
                        WallColor = tile.WallColor
                    };
                }
            }
        }

        // 记录家具位置
        copiedData.Furniture = ScanFurniture(startX, startY, width, height);
    }

    // 扫描家具
    private List<FurnitureData> ScanFurniture(int startX, int startY, int width, int height)
    {
        var furnitureList = new List<FurnitureData>();

        for (int x = startX; x < startX + width; x++)
        {
            for (int y = startY; y < startY + height; y++)
            {
                if (WorldGen.InWorld(x, y))
                {
                    Tile tile = Main.tile[x, y];
                    
                    if (tile.HasTile && Main.tileFrameImportant[tile.TileType])
                    {
                        // 检查是否是家具的左上角
                        TileObjectData data = TileObjectData.GetTileData(tile.TileType, 0);
                        if (data != null)
                        {
                            int originX = x - (tile.TileFrameX / 18) % data.Width;
                            int originY = y - (tile.TileFrameY / 18) % data.Height;

                            if (originX >= startX && originX < startX + width &&
                                originY >= startY && originY < startY + height)
                            {
                                // 只记录左上角位置的家具
                                if (originX == x && originY == y)
                                {
                                    furnitureList.Add(new FurnitureData
                                    {
                                        RelativeX = x - startX,
                                        RelativeY = y - startY,
                                        Type = tile.TileType,
                                        Style = tile.TileFrameX / data.CoordinateWidth
                                    });
                                }
                            }
                        }
                    }
                }
            }
        }

        return furnitureList;
    }

    // 粘贴结构
    public void PasteStructure(int targetX, int targetY)
    {
        if (copiedData == null)
            return;

        // 清理目标区域
        ClearArea(targetX, targetY, copiedData.Width, copiedData.Height);

        // 放置Tile
        for (int x = 0; x < copiedData.Width; x++)
        {
            for (int y = 0; y < copiedData.Height; y++)
            {
                int worldX = targetX + x;
                int worldY = targetY + y;

                if (WorldGen.InWorld(worldX, worldY))
                {
                    TileData tileData = copiedData.Tiles[x][y];

                    if (tileData.HasTile)
                    {
                        WorldGen.PlaceTile(worldX, worldY, tileData.TileType);
                        
                        Main.tile[worldX, worldY].TileColor = tileData.TileColor;
                        Main.tile[worldX, worldY].IsHalfBlock = tileData.IsHalfBlock;
                        Main.tile[worldX, worldY].Slope = tileData.Slope;
                    }
                }
            }
        }

        // 放置Wall
        for (int x = 0; x < copiedData.Width; x++)
        {
            for (int y = 0; y < copiedData.Height; y++)
            {
                int worldX = targetX + x;
                int worldY = targetY + y;

                if (WorldGen.InWorld(worldX, worldY))
                {
                    WallData wallData = copiedData.Walls[x][y];
                    
                    if (wallData.WallType > 0)
                    {
                        WorldGen.PlaceWall(worldX, worldY, wallData.WallType);
                        Main.tile[worldX, worldY].WallColor = wallData.WallColor;
                    }
                }
            }
        }

        // 放置家具
        foreach (var furniture in copiedData.Furniture)
        {
            int worldX = targetX + furniture.RelativeX;
            int worldY = targetY + furniture.RelativeY;

            if (WorldGen.InWorld(worldX, worldY))
            {
                WorldGen.PlaceObject(worldX, worldY, furniture.Type, false, furniture.Style);
            }
        }

        // 同步
        SyncArea(targetX, targetY, copiedData.Width, copiedData.Height);
    }
}
```

## 4. Structure Helper模组分析

Structure Helper是一个专门用于创建和保存建筑结构的模组，是建筑生成开发的重要工具。

**主要功能：**
- **结构选择**：在游戏中选择区域
- **结构保存**：保存为JSON或自定义格式
- **结构加载**：从文件加载结构
- **结构放置**：将结构放置到世界中
- **API接口**：供其他模组调用

### 4.4 自定义Schematic加载器

如果不使用Structure Helper模组，可以自行实现：

```csharp
// 自定义Schematic类
public class SchematicData
{
    public string Name;
    public int Width;
    public int Height;
    public SchematicTile[][] Tiles;
    public SchematicWall[][] Walls;
    public SchematicFurniture[] Furniture;
}

public struct SchematicTile
{
    public ushort Type;
    public short FrameX;
    public short FrameY;
    public byte Color;
}

public struct SchematicWall
{
    public ushort Type;
    public byte Color;
}

public struct SchematicFurniture
{
    public int X;
    public int Y;
    public ushort Type;
    public int Style;
}

// Schematic加载器
public class SchematicLoader
{
    // 从JSON文件加载
    public static SchematicData LoadFromJSON(string path)
    {
        string json = File.ReadAllText(path);
        return JsonConvert.DeserializeObject<SchematicData>(json);
    }

    // 从二进制文件加载（更高效）
    public static SchematicData LoadFromBinary(string path)
    {
        using (var stream = new FileStream(path, FileMode.Open))
        using (var reader = new BinaryReader(stream))
        {
            var data = new SchematicData();
            
            data.Name = reader.ReadString();
            data.Width = reader.ReadInt32();
            data.Height = reader.ReadInt32();

            // 读取Tile数据
            data.Tiles = new SchematicTile[data.Width][];
            for (int x = 0; x < data.Width; x++)
            {
                data.Tiles[x] = new SchematicTile[data.Height];
                for (int y = 0; y < data.Height; y++)
                {
                    data.Tiles[x][y] = new SchematicTile
                    {
                        Type = reader.ReadUInt16(),
                        FrameX = reader.ReadInt16(),
                        FrameY = reader.ReadInt16(),
                        Color = reader.ReadByte()
                    };
                }
            }

            // 读取Wall数据
            data.Walls = new SchematicWall[data.Width][];
            for (int x = 0; x < data.Width; x++)
            {
                data.Walls[x] = new SchematicWall[data.Height];
                for (int y = 0; y < data.Height; y++)
                {
                    data.Walls[x][y] = new SchematicWall
                    {
                        Type = reader.ReadUInt16(),
                        Color = reader.ReadByte()
                    };
                }
            }

            // 读取家具数据
            int furnitureCount = reader.ReadInt32();
            data.Furniture = new SchematicFurniture[furnitureCount];
            for (int i = 0; i < furnitureCount; i++)
            {
                data.Furniture[i] = new SchematicFurniture
                {
                    X = reader.ReadInt32(),
                    Y = reader.ReadInt32(),
                    Type = reader.ReadUInt16(),
                    Style = reader.ReadInt32()
                };
            }

            return data;
        }
    }

    // 保存到JSON
    public static void SaveToJSON(SchematicData data, string path)
    {
        string json = JsonConvert.SerializeObject(data, Formatting.Indented);
        File.WriteAllText(path, json);
    }

    // 保存到二进制
    public static void SaveToBinary(SchematicData data, string path)
    {
        using (var stream = new FileStream(path, FileMode.Create))
        using (var writer = new BinaryWriter(stream))
        {
            writer.Write(data.Name);
            writer.Write(data.Width);
            writer.Write(data.Height);

            // 写入Tile数据
            for (int x = 0; x < data.Width; x++)
            {
                for (int y = 0; y < data.Height; y++)
                {
                    writer.Write(data.Tiles[x][y].Type);
                    writer.Write(data.Tiles[x][y].FrameX);
                    writer.Write(data.Tiles[x][y].FrameY);
                    writer.Write(data.Tiles[x][y].Color);
                }
            }

            // 写入Wall数据
            for (int x = 0; x < data.Width; x++)
            {
                for (int y = 0; y < data.Height; y++)
                {
                    writer.Write(data.Walls[x][y].Type);
                    writer.Write(data.Walls[x][y].Color);
                }
            }

            // 写入家具数据
            writer.Write(data.Furniture.Length);
            foreach (var furniture in data.Furniture)
            {
                writer.Write(furniture.X);
                writer.Write(furniture.Y);
                writer.Write(furniture.Type);
                writer.Write(furniture.Style);
```

```csharp
        }

        // 天花板
        for (int i = x; i < x + width; i++)
        {
            WorldGen.PlaceTile(i, y, wallMaterial);
        }

        // 左墙
        for (int j = y; j < y + height; j++)
        {
            WorldGen.PlaceTile(x, j, wallMaterial);
        }

        // 右墙
        for (int j = y; j < y + height; j++)
        {
            WorldGen.PlaceTile(x + width - 1, j, wallMaterial);
        }
    }

    // 生成背景墙
    private void GenerateBackgroundWall(int x, int y, int width, int height, ushort wallType)
    {
        for (int i = x; i < x + width; i++)
        {
            for (int j = y; j < y + height; j++)

                WorldGen.PlaceWall(i, j, wallType);
            }
        }

        int doorX = x + width / 2;
        int doorY = y + height - 4;  // 门是3格高

        // 清理门空间
        for (int i = doorY; i < doorY + 3; i++)

        }

        // 放置门
        WorldGen.PlaceDoor(doorX, doorY, TileID.ClosedDoor);
    }

    {

        // 工作台
        int tableX = x + 2;
        WorldGen.PlaceObject(tableX, floorY, TileID.WorkBench);

        WorldGen.PlaceObject(chairX, floorY, TileID.Chairs);

        // 床（可选）

        }

            WorldGen.PlaceObject(torch2X, torchY, TileID.Torches);
        }
    }
```

```csharp
public class BuilderWand : ModItem

    public override void SetDefaults()
    {
        Item.width = 24;
        Item.height = 24;
        Item.useStyle = ItemUseStyleID.Swing;

    {
        // 获取目标位置
        int targetX = Player.tileTargetX;
        int targetY = Player.tileTargetY;

        switch (currentMode)
        {
            case PlacementMode.Normal:
                PlaceSingleTile(targetX, targetY, selectedTileType);
                break;

            case PlacementMode.Line:
                PlaceLineTiles(player, targetX, targetY, selectedTileType);
                break;

            case PlacementMode.Wall:

                break;

            case PlacementMode.Platform:
                PlacePlatformTiles(targetX, targetY, selectedTileType);
                break;

            case PlacementMode.Bridge:
                PlaceBridgeTiles(player, targetX, targetY, selectedTileType);
                break;
        }

        return true;
    }

    private void PlaceLineTiles(Player player, int targetX, int targetY, ushort tileType)
    {
        // 基于玩家移动方向放置
        int direction = player.direction;
        int length = 5;  // 连续放置5格

        for (int i = 0; i < length; i++)
        {
            int x = targetX + i * direction;
            if (WorldGen.InWorld(x, targetY))
            {
                WorldGen.PlaceTile(x, targetY, tileType);
            }
        }

        // 同步
        if (Main.netMode == NetmodeID.Server)
        {
            NetMessage.SendTileSquare(-1, targetX, targetY, length, 1);
        }
    }

    // 墙壁模式（2格高）
    private void PlaceWallTiles(int targetX, int targetY, ushort tileType)
    {
        for (int y = targetY; y < targetY + 2; y++)
        {
            if (WorldGen.InWorld(targetX, y))
            {
                WorldGen.PlaceTile(targetX, y, tileType);
            }
        }

        SyncArea(targetX, targetY, 1, 2);
    }

    // 平台模式
    private void PlacePlatformTiles(int targetX, int targetY, ushort tileType)
    {
        // 平台是特殊的Tile，不需要支撑
        WorldGen.PlaceTile(targetX, targetY, TileID.Platforms);
        SyncArea(targetX, targetY, 1, 1);
    }

    // 桥梁模式（自动检测下方是否有支撑）
    private void PlaceBridgeTiles(Player player, int targetX, int targetY, ushort tileType)
    {
        int direction = player.direction;
        int bridgeLength = 10;

        for (int i = 0; i < bridgeLength; i++)
        {
            int x = targetX + i * direction;
            int y = targetY;

            if (WorldGen.InWorld(x, y))
            {
                // 检查下方是否有支撑
                if (!WorldGen.SolidTile(x, y + 1))
                {
                    // 添加支撑柱
                    WorldGen.PlaceTile(x, y + 1, tileType);
                    WorldGen.PlaceTile(x, y + 2, tileType);
                }

                // 放置桥面
                WorldGen.PlaceTile(x, y, TileID.Platforms);
            }
        }

        SyncArea(targetX, targetY, bridgeLength, 3);
    }
```

```csharp
/// <summary>
/// 在地表生成建筑，自动适配地形
/// </summary>
public void GenerateSurfaceBuilding(int startX, int width, int height)
{
    // 1. 找到地表高度
    int[] surfaceHeights = new int[width];
    for (int i = 0; i < width; i++)
    {
        surfaceHeights[i] = FindSurfaceY(startX + i);
    }

    // 2. 计算平均高度（或最低高度）
    int targetY = surfaceHeights.Min();  // 使用最低点作为基准

    // 3. 平整地面
    FlattenGround(startX, width, targetY);

    // 4. 生成建筑
    GenerateBuilding(startX, targetY - height, width, height);
}

/// <summary>
/// 查找地表Y坐标
/// </summary>
private int FindSurfaceY(int x)
{
    // 从地表层开始向下搜索
    int startY = (int)Main.worldSurface;
    
    for (int y = startY; y < Main.maxTilesY; y++)
    {
        if (WorldGen.InWorld(x, y) && WorldGen.SolidTile(x, y))
        {
            return y;
        }
    }
    
    return -1;
}

/// <summary>
/// 平整地面
/// </summary>
private void FlattenGround(int startX, int width, int targetY)
{
    for (int i = startX; i < startX + width; i++)
    {
        int surfaceY = FindSurfaceY(i);

        if (surfaceY == -1)
            continue;

        if (surfaceY > targetY)
        {
            // 填充到目标高度
            for (int y = targetY; y <= surfaceY; y++)
            {
                WorldGen.PlaceTile(i, y, TileID.Dirt);
                WorldGen.PlaceWall(i, y, WallID.Dirt);
            }
        }
        else if (surfaceY < targetY)
        {
            // 移除多余方块
            for (int y = surfaceY; y < targetY; y++)
            {
                WorldGen.KillTile(i, y, false, false, true);
            }
        }
    }
}
```

#### 模式6：智能家具放置

```csharp
/// <summary>
/// 智能家具放置算法
/// </summary>
public class FurniturePlacer
{
    /// <summary>
    /// 根据房间大小自动放置家具
    /// </summary>
    public void AutoPlaceFurniture(int roomX, int roomY, int roomWidth, int roomHeight)
    {
        // 必需家具列表
        var requiredFurniture = GetRequiredFurnitureList();

        // 计算可用放置位置
        var availablePositions = CalculateAvailablePositions(roomX, roomY, roomWidth, roomHeight);

        // 优化放置位置
        foreach (var furniture in requiredFurniture)
        {
            var bestPosition = FindBestPosition(furniture, availablePositions);
            
            if (bestPosition != null)
            {
                PlaceFurniture(furniture, bestPosition);
                availablePositions.Remove(bestPosition);
            }
        }
    }

    /// <summary>
    /// 获取必需家具列表
    /// </summary>
    private List<FurnitureItem> GetRequiredFurnitureList()
    {
        return new List<FurnitureItem>
        {
            new FurnitureItem { Type = TileID.WorkBench, Priority = 1, Width = 2 },
            new FurnitureItem { Type = TileID.Chairs, Priority = 2, Width = 1 },
            new FurnitureItem { Type = TileID.Torches, Priority = 3, Width = 1 },
            // 可选家具
            new FurnitureItem { Type = TileID.Containers, Priority = 4, Width = 2, Optional = true },
            new FurnitureItem { Type = TileID.Beds, Priority = 5, Width = 4, Optional = true }
        };
    }

    /// <summary>
    /// 计算可用放置位置
    /// </summary>
    private List<PlacementPosition> CalculateAvailablePositions(int roomX, int roomY, int roomWidth, int roomHeight)
    {
        var positions = new List<PlacementPosition>();
        int floorY = roomY + roomHeight - 2;  // 家具放在地板上一层

        for (int x = roomX + 1; x < roomX + roomWidth - 1; x++)
        {
            // 检查位置是否可用
            if (IsValidFurniturePosition(x, floorY))
            {
                positions.Add(new PlacementPosition { X = x, Y = floorY });
            }
        }

        return positions;
    }

    /// <summary>
    /// 查找最佳放置位置
    /// </summary>
    private PlacementPosition FindBestPosition(FurnitureItem furniture, List<PlacementPosition> availablePositions)
    {
        // 根据家具类型选择最佳位置
        // 例如：工作台放在角落，椅子靠近工作台，火把分散放置

        switch (furniture.Type)
        {
            case TileID.WorkBench:
                // 工作台放在角落
                return availablePositions.FirstOrDefault(p => p.X <= availablePositions.Min(pos => pos.X) + 2);

            case TileID.Chairs:
                // 椅子靠近工作台
                return availablePositions.FirstOrDefault(p => p.X >= 4 && p.X <= 6);

            case TileID.Torches:
                // 火把放在中间位置
                return availablePositions.FirstOrDefault(p => p.X >= availablePositions.Count / 2);

            case TileID.Containers:
                // 箱子放在角落
                return availablePositions.FirstOrDefault(p => p.X >= availablePositions.Max(pos => pos.X) - 2);

            case TileID.Beds:
                // 床需要足够空间
                return availablePositions.FirstOrDefault(p => 
                    availablePositions.Any(pos => pos.X >= p.X && pos.X <= p.X + furniture.Width));

            default:
                return availablePositions.FirstOrDefault();
        }
    }

    /// <summary>
    /// 检查家具位置是否有效
    /// </summary>
    private bool IsValidFurniturePosition(int x, int y)
    {
        if (!WorldGen.InWorld(x, y))
            return false;

        Tile tile = Main.tile[x, y];
        
        // 位置必须是空的
        if (tile.HasTile)
            return false;

        // 下方必须有支撑
        if (!WorldGen.SolidTile(x, y + 1))
            return false;

        return true;
    }
}
```

```csharp
    /// <summary>
    /// 处理服务器接收的建筑请求
    /// </summary>
    public void HandleBuildingRequest(BinaryReader reader)
    {
        int x = reader.ReadInt32();
        int y = reader.ReadInt32();
        int width = reader.ReadInt32();
        int height = reader.ReadInt32();
        BuildingData building = ReadBuildingData(reader);

        // 验证请求
        if (IsValidBuildingRequest(x, y, width, height))
        {
            PlaceBuildingInternal(x, y, width, height, building);
            NetMessage.SendTileSquare(-1, x, y, width, height);
        }
    }

    /// <summary>
    /// 验证建筑请求是否有效
    /// </summary>
    private bool IsValidBuildingRequest(int x, int y, int width, int height)
    {
        // 检查坐标范围
        if (!WorldGen.InWorld(x, y, 10))
            return false;
        if (!WorldGen.InWorld(x + width, y + height, 10))
            return false;

        // 检查是否与其他结构冲突
        foreach (var structure in GenVars.structures)
        {
            if (structure.Intersects(new Rectangle(x, y, width, height)))
                return false;
        }

        return true;
    }
}
```

## 7. 实现思路总结

### 7.1 建筑生成模组开发要点

| 要点 | 说明 |
|------|------|
| **数据结构设计** | 使用二维数组存储Tile/Wall数据，支持家具位置记录 |
| **放置算法** | 使用WorldGen.PlaceTile/PlaceWall/PlaceObject进行放置 |
| **多人同步** | 使用NetMessage.SendTileSquare同步修改区域 |
| **地形适配** | 实现地表检测和地形平整算法 |
| **家具布局** | 实现智能家具放置算法，确保房屋有效性 |
| **模板系统** | 支持JSON或二进制格式的结构保存/加载 |

### 7.2 核心API调用流程

```
1. 准阶段
   └─> 清理目标区域 (WorldGen.EmptyTile / KillTile)
   └─> 检测地形适配 (FindSurface, FlattenGround)

2. 建造阶段
   └─> 放置框架Tile (WorldGen.PlaceTile)
   └─> 放置背景墙 (WorldGen.PlaceWall)
   └─> 放置家具 (WorldGen.PlaceObject, PlaceDoor)

3. 完成阶段
   └─> 更新Tile帧 (WorldGen.SquareTileFrame)
   └─> 同步到客户端 (NetMessage.SendTileSquare)
   └─> 记录结构位置 (GenVars.structures.AddStructure)
```

### 7.3 NPC房屋有效性检查要点

| 要点 | 要求 |
|------|------|
| **尺寸** | 最少60个Tile，内部至少8x7或10x6 |
| **背景墙** | 必须有玩家放置的墙（天然墙无效） |
| **框架** | 四周必须有实心方块包围 |
| **入口** | 必须有门或平台作为入口 |
| **光源** | 必须有火把、蜡烛等光源 |
| **平坦表面** | 必须有桌子或工作台 |
| **舒适物品** | 必须有椅子或床 |
| **生物群落** | 不能处于腐化/猩红区域 |

### 7.4 建筑生成模组开发建议

1. **模块化设计**
   - 将数据结构、放置算法、同步逻辑分离
   - 使用接口和抽象类提高可扩展性

2. **性能优化**
   - 使用二进制格式存储大型结构
   - 批量Tile操作后一次性同步
   - 避免逐Tile发送网络消息

3. **用户体验**
   - 提供预览功能（放置前显示轮廓）
   - 支持撤销/重做操作
   - 提供多种预设模板

4. **兼容性考虑**
   - 与其他模组的结构不冲突
   - 支持多人游戏环境
   - 处理各种异常情况
